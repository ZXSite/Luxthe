{{- $page := . -}} {{- $hasWidget := false -}} {{- $searchAvailable := partial "helper/search-available.html" $page -}} {{- $isMainContentPage := partial "helper/is-main-content-page.html" $page -}} {{- if $searchAvailable -}} {{- /* Always show widget sidebar if search is available */ -}} {{- $hasWidget = true -}} {{- end -}} {{- if or $page.IsHome (in (slice "section" "taxonomy" "term") $page.Kind) -}} {{- range (default (slice) (index $page.Site.Params.widgets "homepage")) -}} {{- $partialPath := printf "_partials/widget/%s.html" .type -}} {{- if and (templates.Exists $partialPath) (or (ne .type "search") $searchAvailable) -}} {{- $hasWidget = true -}} {{- end -}} {{- end -}} {{- else if $isMainContentPage -}} {{- range (default (slice) (index $page.Site.Params.widgets "page")) -}} {{- $partialPath := printf "_partials/widget/%s.html" .type -}} {{- if and (templates.Exists $partialPath) (ne .type "search") -}} {{- if eq .type "toc" -}} {{- $tocEnabled := default $page.Site.Params.article.toc $page.Params.toc -}} {{- if and $tocEnabled (ge (len $page.TableOfContents) 100) -}} {{- $hasWidget = true -}} {{- end -}} {{- else -}} {{- $hasWidget = true -}} {{- end -}} {{- end -}} {{- end -}} {{- else -}} {{- /* Archives, search, and other special pages get widgets too */ -}} {{- $hasWidget = true -}} {{- end -}} {{- return $hasWidget -}}
